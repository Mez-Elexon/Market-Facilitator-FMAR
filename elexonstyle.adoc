  catalog:
    # Define font families. Make sure these fonts are installed on your system or accessible.
    MySerif:
      normal: Times-Roman # Example, use actual font names
      italic: Times-Italic
      bold: Times-Bold
      bold_italic: Times-BoldItalic

      normal: Courier # Example
